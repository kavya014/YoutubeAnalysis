size=[]
select = [data_CA , data_DE , data_FR , data_GB , data_IN , data_JP , data_KR , data_MX , 
            data_RU , data_US]
countries = ['Great Britan', 'India', 'Russia', 'Mexico', 'Japan',
             'United States', 'France', 'Denmark', 'South Korea', 'Canada']
for i in select:
    total_rows = len(i.index)
    size.append(total_rows)
In [4]:
K=[]
for i in range(0,len(countries)):
    K.append(i)

plt.plot(K,size)
plt.xlabel('Countries' , fontsize =25)
plt.ylabel('No. of Instances',fontsize=25)
plt.xticks(K,countries,rotation=30,horizontalalignment='right')
plt.show()
